        render: function () {
            var regions = this.region.regions;
            var me = this;
            var $element = $(PageTemplate());

            this.$el.append($element);

            _.forEach(_.keys(this.subregions), function (region) {
                if (regions[region]) {
                    me.$el.find('#' + region).replaceWith(regions[region].view.el);
                }
            });
        }
    });
